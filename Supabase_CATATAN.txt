-- Aktifkan RLS dulu kalau belum aktif
ALTER TABLE data_user ENABLE ROW LEVEL SECURITY;

-- Policy untuk SELECT
CREATE POLICY "Allow select for everyone" ON data_user
FOR SELECT
USING (true);

-- Policy untuk INSERT
CREATE POLICY "Allow insert for everyone" ON data_user
FOR INSERT
WITH CHECK (true);

-- Policy untuk UPDATE
CREATE POLICY "Allow update for everyone" ON data_user
FOR UPDATE
USING (true)
WITH CHECK (true);

-- Policy untuk DELETE
CREATE POLICY "Allow delete for everyone" ON data_user
FOR DELETE
USING (true);
